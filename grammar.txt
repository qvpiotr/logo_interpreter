# TOKENS

STRING ::= [a-zA-Z_][a-zA-Z0-9_]*
FLOAT ::= [0-9]*.[0-9]+
INT ::== [0-9]+
TRUE ::== 'true'
FALSE ::== 'false'
LBR: ::== '['
RBR: ::== ']'
LPAR ::== '('
RPAR ::== ')'
QUOTE ::== '"'
COLON ::== ':'
COMMA ::== ','
PLUS ::== '+'
MINUS ::== '-'
TIMES ::== '*'
DIVIDE ::== '/'
POWER ::== '^'
EQUALS ::== '='
GTE ::== '>='
LTE ::== '<='
GT ::== '>'
LT ::== '<'
NE ::== '!='
REPEAT ::== 'repeat'
FORWARD ::== 'fd' | 'forward'
BACK ::== 'bk' | 'back'
RIGHT ::== 'rt' | 'right'
LEFT ::== 'lt' | 'left'
PENUP ::== 'pu' | 'penup'
PENDOWN ::== 'pd' | 'pendown'
PRINT ::== 'print'
MAKE ::== 'make'
IF ::== 'if'
IFELSE ::== 'ifelse'
WHILE ::== 'while' | 'until'
DO ::== 'do'
TO ::== 'to'
END ::== 'end'
REPCOUNT ::== 'repcount'
SHOW ::== 'st' | 'showturtle'
HIDE ::== 'ht' | 'hideturtle'
SETX ::== 'setx'
SETY ::== 'sety'
SETXY ::== 'setxy' | 'setpos'
SETH ::== 'setheading' | 'seth'
SETPC ::== 'setpencolor' | 'setcolor' | 'setpc'
SETBC ::== 'setbgcolor'
SETPS ::== 'setpensize' | 'setpenwidth | 'setpw'
HOME ::== 'home'
SPEED ::== 'speed'
CLEAN ::== 'clean'
RESET ::== 'reset' | 'cs' | 'clearscreen'
RANDOM ::== 'random'


    # GRAMMAR

    statement_list ::== statement_list statement | statement

    statement ::== turtle_instruction
                | repeat_statement
                | if_statement
                | ifelse_statement
                | while_statement
                | dowhile_statement
                | variable_declaration
                | procedure_definition
                | procedure_call
                | print_statement

    turtle_instruction ::== SETXY LBR expression COMMA expression RBR
                        | FORWARD expression
                        | BACK expression
                        | RIGHT expression
                        | LEFT expression
                        | SETX expression
                        | SETY expression
                        | SETH expression
                        | SPEED expression
                        | SETPS expression
                        | SETPC word
                        | SETBC word
                        | PENUP
                        | PENDOWN
                        | SHOW
                        | HIDE
                        | HOME
                        | CLEAN
                        | RESET

    repeat_statement ::== REPEAT expression LBR statement_list RBR

    if_statement ::== IF condition LBR statement_list RBR

    ifelse_statement ::== IFELSE condition LBR statement_list RBR LBR statement_list RBR

    while_statement ::== WHILE condition LBR statement_list RBR

    dowhile_statement ::== DO LBR statement_list RBR WHILE condition

    variable_declaration ::== MAKE word expression

    procedure_definition ::== TO STRING LBR parameter_list RBR statement_list END

    parameter_list ::== parameter_list COMMA parameter
                    | parameter

    parameter ::== name | empty

    procedure_call ::== STRING LBR expression_list RBR

    expression_list ::== expression_list COMMA expression
                    | expression 
                    | empty

    print_statement ::== PRINT word | PRINT expression

    expression ::== INT
                | FLOAT 
                | REPCOUNT
                | name
                | MINUS expression
                | expression PLUS expression
                | expression MINUS expression
                | expression TIMES expression
                | expression DIVIDE expression
                | expression POWER expression
                | LPAR expression RPAR 

    condition ::== TRUE
                | FALSE
                | expression GT expression
                | expression LT expression
                | expression GTE expression
                | expression LTE expression
                | expression EQUALS expression
                | expression NE expression

    word ::== QUOTE STRING

    name ::== COLON STRING

    empty :
